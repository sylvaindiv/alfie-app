- Width: 150px
- Height: 150px
- Border-radius: 8px

- Color: white
- Padding: 4px 8px
- Border-radius: 4px
- Font-size: 10px

- Background: rgba(0,0,0,0.6)
- Color: white
- Size: 32x32px
- Border-radius: 4px
